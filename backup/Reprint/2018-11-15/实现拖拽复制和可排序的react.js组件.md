            if (sort &lt; dropedSort) {
                data.map(item =&gt; {
                    if (item[codeKey] === code) {
                        item[sortKey] = dropedSort;
                    } else if (item[sortKey] &gt; sort &amp;&amp; item[sortKey] &lt; dropedSort + 1) {
                        item[sortKey]--;
                    }
                    return item;
                });

                    if (item[codeKey] === code) {
                        item[sortKey] = dropedSort;

                        item[sortKey]++;

                    return item;
                });

            let objDragedItem = JSON.parse(dragedItem);
